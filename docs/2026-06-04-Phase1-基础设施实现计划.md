```python
# src/accounting_mapper.py
"""
中国会计准则 → IFRS/US GAAP 科目映射引擎

参考：中国《企业会计准则》科目表 + AKShare/ZVT 科目体系
"""
from typing import Optional
import pandas as pd
import difflib


class AccountingMapper:
    """
    将 MinerU / Excel 解析出的中文科目名称，
    映射为 FinanceToolkit 可识别的标准英文字段名。
    """

    # ────── 标准科目字典 ──────
    # 格式：英文字段名 → [中文标准名, 别名1, 别名2, ...]
    STANDARD_MAP: dict[str, list[str]] = {
        # ── 资产负债表科目 ──
        "cash_and_equivalents": ["货币资金", "现金及银行存款", "现金"],
        "accounts_receivable": ["应收账款", "应收帐款", "应收票据", "应收款项"],
        "inventory": ["存货", "库存商品", "原材料"],
        "prepaid_expenses": ["预付账款", "预付款项", "预付帐款"],
        "other_receivables": ["其他应收款", "其它应收款"],
        "total_current_assets": ["流动资产合计", "流动资产总计"],
        "fixed_assets": ["固定资产", "固定资产原值"],
        "intangible_assets": ["无形资产"],
        "long_term_investments": ["长期股权投资", "长期投资"],
        "total_assets": ["资产总计", "资产合计", "总资产"],
        "short_term_borrowings": ["短期借款", "短期借贷"],
        "accounts_payable": ["应付账款", "应付帐款", "应付票据", "应付款项"],
        "advances_from_customers": ["预收账款", "预收款项", "预收帐款"],
        "wages_payable": ["应付职工薪酬", "应付工资", "应付薪酬"],
        "tax_payable": ["应交税费", "应交税金", "应缴税费"],
        "other_payables": ["其他应付款", "其它应付款"],
        "total_current_liabilities": ["流动负债合计", "流动负债总计"],
        "long_term_borrowings": ["长期借款", "长期借贷", "长期应付款"],
        "total_liabilities": ["负债合计", "负债总计"],
        "paid_in_capital": ["实收资本", "股本"],
        "retained_earnings": ["未分配利润", "留存收益"],
        "total_equity": ["所有者权益合计", "股东权益合计", "权益合计"],
        "total_liabilities_and_equity": ["负债及所有者权益总计", "负债与股东权益总计"],

        # ── 利润表科目 ──
        "revenue": ["营业收入", "主营业务收入", "销售收入", "营收"],
        "cost_of_goods_sold": ["营业成本", "主营业务成本", "销售成本"],
        "tax_and_surcharge": ["税金及附加", "营业税金及附加"],
        "selling_expenses": ["销售费用", "营业费用", "销售及管理费用"],
        "administrative_expenses": ["管理费用", "行政管理费用"],
        "rd_expenses": ["研发费用", "研发支出", "研究开发费"],
        "financial_expenses": ["财务费用", "利息费用"],
        "other_income": ["其他收益", "其它收益", "政府补助"],
        "investment_income": ["投资收益"],
        "non_operating_income": ["营业外收入"],
        "non_operating_expense": ["营业外支出"],
        "asset_impairment_loss": ["资产减值损失"],
        "credit_impairment_loss": ["信用减值损失"],
        "income_tax_expense": ["所得税费用", "所得税"],
        "net_profit": ["净利润", "税后利润"],
        "total_profit": ["利润总额", "税前利润"],

        # ── 现金流量表科目 ──
        "operating_cash_flow": ["经营活动产生的现金流量净额", "经营活动现金流净额"],
        "investing_cash_flow": ["投资活动产生的现金流量净额", "投资活动现金流净额"],
        "financing_cash_flow": ["筹资活动产生的现金流量净额", "筹资活动现金流净额"],
        "capital_expenditure": ["购建固定资产无形资产支付的现金", "资本支出"],
    }

    # ────── 汇总行关键词 ──────
    SUMMARY_KEYWORDS = {"合计", "总计", "小计", "总资产", "总负债", "total"}

    # ────── 模糊匹配阈值 ──────
    FUZZY_THRESHOLD = 0.75

    def __init__(self, custom_map: Optional[dict] = None):
        """
        Args:
            custom_map: 自定义映射字典，格式同 STANDARD_MAP，
                        会追加/覆盖默认映射
        """
        self._map = dict(self.STANDARD_MAP)
        if custom_map:
            self._map.update(custom_map)

        # 构建反向索引：中文名 → 英文字段
        self._zh_to_en: dict[str, str] = {}
        for en_key, zh_names in self._map.items():
            for zh in zh_names:
                self._zh_to_en[zh.strip()] = en_key

        # 构建模糊匹配的已知中文名列表
        self._known_zh_names = list(self._zh_to_en.keys())

    def lookup(self, zh_account_name: str) -> str:
        """
        精确匹配 + 模糊匹配兜底，返回英文字段名。
        未命中时返回原始中文名的清洗版本。
        """
        name = str(zh_account_name).strip()

        # 1. 精确匹配
        if name in self._zh_to_en:
            return self._zh_to_en[name]

        # 2. 编辑距离模糊匹配
        best_match = difflib.get_close_matches(
            name, self._known_zh_names, n=1, cutoff=self.FUZZY_THRESHOLD
        )
        if best_match:
            return self._zh_to_en[best_match[0]]

        # 3. 兜底：返回Unicode安全的键名
        safe_name = name.replace(" ", "_").replace("/", "_")
        return f"unmapped_{safe_name}"

    def is_summary_row(self, zh_account_name: str) -> bool:
        """判断是否为汇总行"""
        name = str(zh_account_name).strip()
        return any(kw in name for kw in self.SUMMARY_KEYWORDS)

    def map_to_standard(
        self,
        df: pd.DataFrame,
        acct_col: str = "科目名称",
        value_col: str = "期末余额",
    ) -> pd.DataFrame:
        """
        将中文科目DataFrame转换为标准格式。

        输入示例：
            科目名称    期末余额
            货币资金    1000000
            营业收入    5000000

        输出示例：
            cash_and_equivalents  revenue  ...  original_name  is_summary_row
            1000000              5000000  ...  货币资金        False

        Args:
            df: 中文财报DataFrame
            acct_col: 科目名称所在列名
            value_col: 数值所在列名

        Returns:
            标准化的DataFrame，列为英文字段名 + original_name + is_summary_row
        """
        records: dict[str, list] = {
            "original_name": [],
            "english_field": [],
            "is_summary_row": [],
            value_col: [],
        }

        for _, row in df.iterrows():
            zh_name = str(row[acct_col]).strip()
            if not zh_name or zh_name.lower() in ("nan", "none"):
                continue

            en_field = self.lookup(zh_name)
            is_sum = self.is_summary_row(zh_name)

            records["original_name"].append(zh_name)
            records["english_field"].append(en_field)
            records["is_summary_row"].append(is_sum)
            records[value_col].append(row[value_col])

        # 转换为宽表：每列一个英文字段
        result = pd.DataFrame(records)
        pivot = result.pivot_table(
            index=None,
            columns="english_field",
            values=value_col,
            aggfunc="first",
        )
        pivot = pivot.reset_index(drop=True)

        # 附加元数据列
        pivot["_original_names"] = [result["original_name"].tolist()]
        pivot["_summary_mask"] = [result["is_summary_row"].tolist()]

        return pivot
```

```python
# src/quality_checker.py
"""
财务数据质量校验层

在指标计算前自动检测数据问题，按严重级别分类：
- block: 阻断级，需用户确认或修正数据
- warn:  警告级，照常计算但报告中标注
- ok:    通过
"""
from dataclasses import dataclass, field
import pandas as pd


@dataclass
class CheckResult:
    check_name: str
    passed: bool
    severity: str  # "block" | "warn" | "ok"
    detail: str = ""


class QualityChecker:
    """财务数据勾稽关系与一致性校验引擎"""

    # 资产负债平衡允许的最大差异比例
    BALANCE_TOLERANCE = 0.05

    # 表间勾稽允许的最大差异比例
    INTER_STATEMENT_TOLERANCE = 0.10

    def check_balance_sheet_balance(self, df: pd.DataFrame) -> CheckResult:
        """
        资产负债表平衡校验：资产总计 ≈ 负债合计 + 所有者权益合计
        """
        if not all(c in df.columns for c in ["total_assets", "total_liabilities", "total_equity"]):
            return CheckResult(
                check_name="资产负债表平衡",
                passed=True,  # 数据不全不阻断，仅跳过
                severity="ok",
                detail="未检测到完整的资产/负债/权益数据，跳过平衡校验",
            )

        assets = df["total_assets"].iloc[0]
        liabilities = df["total_liabilities"].iloc[0]
        equity = df["total_equity"].iloc[0]

        diff = abs(assets - (liabilities + equity))
        diff_ratio = diff / max(assets, 1)

        if diff_ratio > self.BALANCE_TOLERANCE:
            return CheckResult(
                check_name="资产负债表平衡",
                passed=False,
                severity="block",
                detail=(
                    f"资产总计 ({assets:,.2f}) ≠ 负债合计 ({liabilities:,.2f}) "
                    f"+ 所有者权益合计 ({equity:,.2f})，"
                    f"差异: {diff:,.2f} ({diff_ratio:.1%})"
                ),
            )

        return CheckResult(
            check_name="资产负债表平衡",
            passed=True,
            severity="ok",
            detail=f"平衡，差异 {diff:,.2f} ({diff_ratio:.4%})",
        )

    def check_negative_values(self, df: pd.DataFrame) -> CheckResult:
        """
        科目异常值检测：
        - 货币资金 < 0 → block
        - 存货 < 0 → block
        - 应收账款 < 0 → warn（可能是预收账款重分类）
        - 应付账款 < 0 → warn（可能是预付账款重分类）
        """
        negative_items = []
        blocking_items = []

        # 阻断级：这些科目不可能为负
        blocking_fields = {
            "cash_and_equivalents": "货币资金",
            "inventory": "存货",
            "fixed_assets": "固定资产",
        }
        for col, label in blocking_fields.items():
            if col in df.columns and (df[col] < 0).any():
                val = df[col].iloc[0]
                blocking_items.append(f"{label}为负 ({val:,.2f})")

        # 警告级：为负可能有合理的重分类理由
        warning_fields = {
            "accounts_receivable": "应收账款",
            "accounts_payable": "应付账款",
        }
        for col, label in warning_fields.items():
            if col in df.columns and (df[col] < 0).any():
                val = df[col].iloc[0]
                negative_items.append(f"{label}为负 ({val:,.2f})，可能需要重分类")

        if blocking_items:
            return CheckResult(
                check_name="科目异常值",
                passed=False,
                severity="block",
                detail="; ".join(blocking_items),
            )

        if negative_items:
            return CheckResult(
                check_name="科目异常值",
                passed=False,
                severity="warn",
                detail="; ".join(negative_items),
            )

        return CheckResult(
            check_name="科目异常值",
            passed=True,
            severity="ok",
            detail="无异常负值科目",
        )

    def check_logic_consistency(self, df: pd.DataFrame) -> CheckResult:
        """
        科目逻辑一致性检测：
        - 营业收入 = 0 但净利润 ≠ 0 → block
        """
        if "revenue" in df.columns and "net_profit" in df.columns:
            revenue = df["revenue"].iloc[0]
            net_profit = df["net_profit"].iloc[0]
            if revenue == 0 and net_profit != 0:
                return CheckResult(
                    check_name="科目逻辑一致性",
                    passed=False,
                    severity="block",
                    detail="营业收入为0但净利润不为0，数据存在逻辑错误",
                )

        return CheckResult(
            check_name="科目逻辑一致性",
            passed=True,
            severity="ok",
            detail="通过",
        )

    def check_extreme_values(self, df: pd.DataFrame) -> CheckResult:
        """
        极值合理性检测：
        - 资产负债率 > 100% → warn
        - 毛利率 > 95% → warn
        """
        warnings = []

        # 资产负债率 = 总负债 / 总资产
        if all(c in df.columns for c in ["total_assets", "total_liabilities"]):
            assets = df["total_assets"].iloc[0]
            liabilities = df["total_liabilities"].iloc[0]
            if assets > 0:
                debt_ratio = liabilities / assets
                if debt_ratio > 1.0:
                    warnings.append(f"资产负债率 {debt_ratio:.1%} 超过100%，已资不抵债")

        # 毛利率 = (营收 - 营业成本) / 营收
        if all(c in df.columns for c in ["revenue", "cost_of_goods_sold"]):
            revenue = df["revenue"].iloc[0]
            cogs = df["cost_of_goods_sold"].iloc[0]
            if revenue > 0:
                gross_margin = (revenue - cogs) / revenue
                if gross_margin > 0.95:
                    warnings.append(f"毛利率 {gross_margin:.1%} 超过95%，请确认数据真实")

        if warnings:
            return CheckResult(
                check_name="极值合理性",
                passed=False,
                severity="warn",
                detail="; ".join(warnings),
            )

        return CheckResult(
            check_name="极值合理性",
            passed=True,
            severity="ok",
            detail="通过",
        )

    def run_all_checks(self, df: pd.DataFrame) -> list[CheckResult]:
        """运行全部校验规则，返回结果列表"""
        results = [
            self.check_balance_sheet_balance(df),
            self.check_negative_values(df),
            self.check_logic_consistency(df),
            self.check_extreme_values(df),
        ]
        return results
```

```python
# src/excel_parser.py
"""
Excel 财务报表解析引擎

MVP 支持格式：第一列为科目名称，后续列为数值。
自动识别科目列（"科目名称"/"项目"/"项 目"等）和数值列（"期末余额"/"本期金额"/"金额"等）。
"""
import pandas as pd


# 常见的科目列名
ACCT_COLUMN_CANDIDATES = [
    "科目名称", "项目", "项    目", "科目", "account",
    "报表项目", "account_name", "item",
]

# 常见的数值列名
VALUE_COLUMN_CANDIDATES = [
    "期末余额", "本期金额", "金额", "余额", "期末数",
    "本期数", "本年累计", "value", "amount",
    "年末数", "年初数",
]


def _detect_columns(df: pd.DataFrame) -> tuple[str, str]:
    """自动检测科目列和数值列"""
    acct_col = None
    value_col = None

    for col in df.columns:
        col_str = str(col).strip()
        if col_str in ACCT_COLUMN_CANDIDATES and acct_col is None:
            acct_col = col
        elif col_str in VALUE_COLUMN_CANDIDATES and value_col is None:
            value_col = col

    # 兜底：第一列做科目，第二列做数值
    if acct_col is None:
        acct_col = df.columns[0]
    if value_col is None:
        numeric_cols = df.select_dtypes(include=["number"]).columns
        if len(numeric_cols) > 0:
            value_col = numeric_cols[0]
        else:
            value_col = df.columns[1] if len(df.columns) > 1 else df.columns[0]

    return str(acct_col), str(value_col)


def parse_financial_excel(file_path: str) -> pd.DataFrame:
    """
    解析上传的 Excel 财务报表，返回统一格式 DataFrame。

    输出列：
        - 科目名称 (原始中文)
        - 金额 (数值)
    """
    df = pd.read_excel(file_path)

    # 去除全空行
    df = df.dropna(how="all")

    acct_col, value_col = _detect_columns(df)

    result = df[[acct_col, value_col]].copy()
    result.columns = ["科目名称", "金额"]

    # 数值清洗
    result["金额"] = pd.to_numeric(result["金额"], errors="coerce")
    result = result.dropna(subset=["科目名称", "金额"])

    # 去除符号前缀 "减：" "其中：" 等
    result["科目名称"] = result["科目名称"].str.replace(
        r"^[一二三四五六七八九十]+[、.]\s*", "", regex=True
    )
    result["科目名称"] = result["科目名称"].str.replace(
        r"^(减：|其中：|加：)", "", regex=True
    )
    result["科目名称"] = result["科目名称"].str.strip()

    return result
```

```python
            for r in warns:
                st.warning(f"🟡 **{r.check_name}**: {r.detail}")
        if oks:
            for r in oks:

            st.balloons()
            st.success("数据质量全部通过！✅")

    st.divider()
    st.caption(f"分析目标：{analysis_goal} | 数据行数：{len(raw_df)} | 识别科目：{len(mapped_df.columns)}")
```
